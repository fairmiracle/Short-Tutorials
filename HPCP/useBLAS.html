15 #define Malloc(type,n) (type *)malloc((n)*sizeof(type))

16 void naivemv(double *A, double *x, double *y, int m, int n);
17 
18 int main(int argc, char const *argv[])
19 {
20 	int n = 10000;
21 	double *A = Malloc(double,n*n);
22 	double *x = Malloc(double,n);
23 	double *y1 = Malloc(double,n);
24 	double *y2 = Malloc(double,n);
25 
26 	/* random generation */ 
27 	srand(time(NULL));
28 	for(int i = 0; i < n; ++i)
29 	{
30 		x[i] = (double)rand() / (double)RAND_MAX;
31 		for(int j = 0; j < n; ++j)
32 			A[i+j*n] = (double)rand() / (double)RAND_MAX;
33 	}
34 
35 	/* call naive implementation */
36 	clock_t tic = clock();
37 	naivemv(A, x, y1, n, n);
38 	double cpu_time_used1 = (double)(clock() - tic) / CLOCKS_PER_SEC;
39 	printf("Naive cpu time: %f\n", cpu_time_used1);
40 
41 	/* call CBLAS Level2 */
42 	tic = clock();
43 	cblas_dgemv(CblasColMajor, CblasNoTrans, n, n, 1, A, n, x, 1, 0, y2, 1);
44 	double cpu_time_used2 = (double)(clock() - tic) / CLOCKS_PER_SEC;
45 	printf("CBLAS cpu time: %f\n", cpu_time_used2);
46 
47 	double *diffy = Malloc(double,n);
48 	memcpy(diffy, y1, n*sizeof(double));
49 	cblas_daxpy(n, -1, y2, 1, diffy, 1);
50 	/* call CBLAS Level1 */
51 	printf("Gap ||y1-y2|| between two methods: %f\n", cblas_ddot(n, diffy, 1, diffy, 1));
52 
53 	/* Clear memory */
54 	free(A);
55 	free(x);
56 	free(y1);
57 	free(y2);
58 	free(diffy);
59 	return EXIT_SUCCESS;
60 }

62 /* Naive implementation of matrix-vector production
63 *  y = A*x, as y_i = \sum_jA_{ij}*x_j
64 */
65 void naivemv(double *A, double *x, double *y, int m, int n)
66 {
67 	for (int i = 0; i < m; ++i)
68 	{
69 		double rowSum = 0;
70 		for (int j = 0; j < n; ++j)
71 		{
72 			rowSum += A[i+j*n]*x[j];
73 		}
74 		y[i] = rowSum;
75 	}
76 }
