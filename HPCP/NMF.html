 63 //A:m*n,column-major; B=A'
 64 void transpose(double *A, double *B, int m, int n){
 65 	int k = 0;
 66 	for( int i = 0; i < m; i++){
 67 		for (int j = 0; j < n; j++)
 68 			B[k++] = A[i+j*m];
 69 	}		
 70 }

 72 //W:m*n, H:n*k, V:m*k, grad:n*k
 73 //At current, all the input matrices are in main memory, and all the out matrices should be in main memory as well.
 74 void nlssubprob(double *V, double *W, double *Hinit, int m, int n, int k, double tol, int maxiter, double *H, double *grad, int *ite){
 75 	//H = Hinit; WtV = W'*V; WtW = W'*W; 
 76 	memcpy(H, Hinit, n*k*sizeof(double));
 77 	
 78 	cublasHandle_t handle;
 79         cublasStatus_t status;
 80 	status = cublasCreate(&handle);CUBLAS_ERROR_CHECK(status);
 81 
 82 	double alpha = 1;
 83 	double beta = 0.1;
 84 
 85 	double alpha1 = 1;
 86 	double beta1 = 0;
 87 
 88 	double* d_grad;
 89 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_grad, k*n*sizeof(double)));
 90 	CUDA_ERROR_CHECK(cudaMemcpy(d_grad, grad, k*n*sizeof(double), cudaMemcpyHostToDevice));
 91 
 92 	double* d_H;
 93 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_H, k*n*sizeof(double)));
 94 	CUDA_ERROR_CHECK(cudaMemcpy(d_H, H, k*n*sizeof(double), cudaMemcpyHostToDevice));
 95 
 96 	double* d_W;
 97 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_W, m*n*sizeof(double)));
 98 	CUDA_ERROR_CHECK(cudaMemcpy(d_W, W, m*n*sizeof(double), cudaMemcpyHostToDevice));
 99 
100 	double* d_V;
101 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_V, k*m*sizeof(double)));
102 	CUDA_ERROR_CHECK(cudaMemcpy(d_V, V, k*m*sizeof(double), cudaMemcpyHostToDevice));
103 	
104 	double *d_WtV = 0;
105 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_WtV, n*k*sizeof(double)));
106 	CUDA_ERROR_CHECK(cudaMemset(d_WtV, 0, n*k*sizeof(double)));
107 	status = cublasDgemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, n, k, m, &alpha1, d_W, m, d_V, m, &beta1, d_WtV, n);CUBLAS_ERROR_CHECK(status);
108 	
109 	double *d_WtW = 0;
110 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_WtW, n*n*sizeof(double)));
111 	CUDA_ERROR_CHECK(cudaMemset(d_WtW, 0, n*n*sizeof(double)));
112 
113 	status = cublasDgemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, n, n, m, &alpha1, d_W, m, d_W, m, &beta1, d_WtW, n);CUBLAS_ERROR_CHECK(status);
114 	
115 	double *Hn = 0;	
116 	Hn = (double *)malloc(n*k*sizeof(double));
117 	memset(Hn, 0, n*k*sizeof(double));
118 	
119 	double *d_Hn = 0;	
120 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_Hn, k*n*sizeof(double)));
121 	CUDA_ERROR_CHECK(cudaMemset(d_Hn, 0, k*n*sizeof(double)));
122 
123 	double *d_d = 0;
124 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_d, k*n*sizeof(double)));
125 	CUDA_ERROR_CHECK(cudaMemset(d_d, 0, k*n*sizeof(double)));
126 	
127 	double *d_WtWd = 0;
128 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_WtWd, k*n*sizeof(double)));
129 	CUDA_ERROR_CHECK(cudaMemset(d_WtWd, 0, k*n*sizeof(double)));
130 	
131 	double *d_Hp = 0;	
132 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_Hp, k*n*sizeof(double)));
133 	CUDA_ERROR_CHECK(cudaMemset(d_Hp, 0, k*n*sizeof(double)));
134 	
135 	double *d_Hnpp = 0;
136 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_Hnpp, k*n*sizeof(double)));
137 	CUDA_ERROR_CHECK(cudaMemset(d_Hnpp, 0, k*n*sizeof(double)));
138 	
139 	double *tmpvec = 0;		
140 	tmpvec = (double *)malloc(n*k*sizeof(double));
141 	memset(tmpvec, 0, n*k*sizeof(double));
142 
143 	double *d_tmpvec = 0;		
144 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_tmpvec, k*n*sizeof(double)));
145 	CUDA_ERROR_CHECK(cudaMemset(d_tmpvec, 0, k*n*sizeof(double)));
146 
147 	int iter = 0;
148 	for ( iter = 1; iter <= maxiter; iter++){
149 		//grad = WtW*H - WtV;
150 		CUDA_ERROR_CHECK(cudaMemcpy(d_grad, d_WtV, k*n*sizeof(double), cudaMemcpyDeviceToDevice));
151 		beta1 = -1;
152 		alpha1 = 1;
153 		status = cublasDgemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, n, k, n, &alpha1, d_WtW, n, d_H, n, &beta1, d_grad, n); CUBLAS_ERROR_CHECK(status);
154 		CUDA_ERROR_CHECK(cudaMemcpy(grad, d_grad, k*n*sizeof(double), cudaMemcpyDeviceToHost));
155 
156 		//projgrad = norm(grad(grad < 0 | H >0));
157 
158 		memset(tmpvec, 0, n*k*sizeof(double));
159 		int ii = 0;
160 		for (int i = 0; i < n*k; i++){
161 			if (grad[i] < 0 || H[i] > 0 )
162 				tmpvec[ii++] = grad[i];
163 		}
164 		CUDA_ERROR_CHECK(cudaMemcpy(d_tmpvec, tmpvec, ii*sizeof(double), cudaMemcpyHostToDevice));
165 		double projgrad = 0;
166 		status = cublasDnrm2(handle, ii, d_tmpvec, 1, &projgrad); CUBLAS_ERROR_CHECK(status);
167 		if (projgrad < tol)
168 			break;
169 		bool decr_alpha = true;
170 		for (int inner_iter = 1; inner_iter <= 20; inner_iter++){
171 			//Hn = max(H - alpha*grad, 0); d = Hn-H;
172 			CUDA_ERROR_CHECK(cudaMemcpy(d_Hn, d_H, n*k*sizeof(double), cudaMemcpyDeviceToDevice));
173 			alpha1 = -alpha;
174 			status = cublasDaxpy(handle, n*k, &alpha1, d_grad, 1, d_Hn, 1);CUBLAS_ERROR_CHECK(status);
175 			CUDA_ERROR_CHECK(cudaMemcpy(Hn, d_Hn, n*k*sizeof(double), cudaMemcpyDeviceToHost));
176 			for (int i = 0; i < n*k; i++){
177 				if (Hn[i] < 0)
178 					Hn[i] = 0;
179 			}
180 			CUDA_ERROR_CHECK(cudaMemcpy(d_Hn, Hn, n*k*sizeof(double), cudaMemcpyHostToDevice));
181 			CUDA_ERROR_CHECK(cudaMemcpy(d_d, d_Hn, n*k*sizeof(double), cudaMemcpyDeviceToDevice));
182 			alpha1 = -1;
183 			status = cublasDaxpy(handle, n*k, &alpha1, d_H, 1, d_d, 1);CUBLAS_ERROR_CHECK(status);
184 			
185 			//gradd=sum(sum(grad.*d)); dQd = sum(sum((WtW*d).*d));
186 			double gradd = 0;
187 			status = cublasDdot (handle, k*n, d_grad, 1, d_d, 1, &gradd);CUBLAS_ERROR_CHECK(status);
188 			alpha1 = 1;
189 			beta1 = 0;
190 			status = cublasDgemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, n, k, n, &alpha1, d_WtW, n, d_d, n, &beta1, d_WtWd, n);CUBLAS_ERROR_CHECK(status);
191 
192 			double dQd = 0;
193 			status = cublasDdot (handle, k*n, d_WtWd, 1, d_d, 1, &dQd);CUBLAS_ERROR_CHECK(status);CUBLAS_ERROR_CHECK(status);
194 
195 			bool suff_decr = 0.99*gradd + 0.5*dQd < 0;
196 			//bool decr_alpha = true;
197 			if (inner_iter==1){
198 				decr_alpha = ~suff_decr; 
199 				CUDA_ERROR_CHECK(cudaMemcpy(d_Hp, d_H, k*n*sizeof(double), cudaMemcpyDeviceToDevice));
200 			}
201 			if(decr_alpha){
202 				if(suff_decr){
203 					CUDA_ERROR_CHECK(cudaMemcpy(d_H, d_Hn, k*n*sizeof(double), cudaMemcpyDeviceToDevice));
204 					break;
205 				}
206 				else
207 					alpha = alpha * beta;
208 			}
209 			else{
210 				CUDA_ERROR_CHECK(cudaMemcpy(d_Hnpp, d_Hn, n*k*sizeof(double), cudaMemcpyDeviceToDevice));
211 				alpha1 = -1; 
212 				status = cublasDaxpy(handle, n*k, &alpha1, d_Hp, 1, d_Hnpp, 1);CUBLAS_ERROR_CHECK(status);
213 				double result  = 0;
214 				status = cublasDnrm2(handle, n*k, d_Hnpp, 1, &result);CUBLAS_ERROR_CHECK(status);
215 				if(~suff_decr || result == 0){
216 					CUDA_ERROR_CHECK(cudaMemcpy(d_H, d_Hp, k*n*sizeof(double), cudaMemcpyDeviceToDevice));
217 					break;
218 				}
219 				else{
220 					alpha = alpha/beta;
221 					CUDA_ERROR_CHECK(cudaMemcpy(d_Hp, Hn, k*n*sizeof(double), cudaMemcpyHostToDevice));
222 				}
223 			}
224 		}
225 		CUDA_ERROR_CHECK(cudaMemcpy(H, d_H, n*k*sizeof(double), cudaMemcpyDeviceToHost));
226 	}
227 
228 	
229 	*ite = iter;
230 	if (*ite==maxiter)
231 		printf("Max iter in nlssubprob\n");
232 
233 	free(Hn);
234 	free(tmpvec);
235 
236 	CUDA_ERROR_CHECK(cudaFree(d_grad));
237 	CUDA_ERROR_CHECK(cudaFree(d_H));
238 	CUDA_ERROR_CHECK(cudaFree(d_W));
239 	CUDA_ERROR_CHECK(cudaFree(d_V));
240 	CUDA_ERROR_CHECK(cudaFree(d_WtV));
241 	CUDA_ERROR_CHECK(cudaFree(d_WtW));
242 	CUDA_ERROR_CHECK(cudaFree(d_Hn));
243 	CUDA_ERROR_CHECK(cudaFree(d_d));
244 	CUDA_ERROR_CHECK(cudaFree(d_WtWd));
245 	CUDA_ERROR_CHECK(cudaFree(d_Hp));
246 	CUDA_ERROR_CHECK(cudaFree(d_Hnpp));
247 	CUDA_ERROR_CHECK(cudaFree(d_tmpvec));
248 
249 	status = cublasDestroy(handle);CUBLAS_ERROR_CHECK(status);
250 	
251 }

253 //NMF:V=W*H,W:m*n, H:n*k, V:m*k
254 //stick to BLAS, column-major
255 //At current, all the input matrices are in main memory, and all the out matrices should be in main memory as well.
256 //We need to transfer the input into device memory and the result out of it.
257 void NMF(double *V, double *Winit,double *Hinit, int m, int n, int k, double tol, double timelimit, int maxiter, double *W, double *H)
258 {
259 	
260 	memcpy(W,Winit,m*n*sizeof(double));
261 	memcpy(H,Hinit,n*k*sizeof(double));
262 
263     cublasHandle_t handle;
264     cublasStatus_t status;
265 	status = cublasCreate(&handle);CUBLAS_ERROR_CHECK(status);
266 	
267 	double* d_V;
268 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_V, k*m*sizeof(double)));
269 	CUDA_ERROR_CHECK(cudaMemcpy(d_V, V, k*m*sizeof(double), cudaMemcpyHostToDevice));
270 
271 	//W = Winit; H = Hinit; initt = cputime;
272 	double* d_W;
273 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_W, m*n*sizeof(double)));
274 	CUDA_ERROR_CHECK(cudaMemcpy(d_W, Winit, m*n*sizeof(double), cudaMemcpyHostToDevice));
275 	
276 	double* d_H;
277 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_H, k*n*sizeof(double)));
278 	CUDA_ERROR_CHECK(cudaMemcpy(d_H, Hinit, k*n*sizeof(double), cudaMemcpyHostToDevice));
279 	
280 
281 	clock_t initt = time(NULL);
282 	
283 	//gradW = W*(H*H') - V*H'; gradH = (W'*W)*H - W'*V;
284 	double *d_HHt = 0;
285 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_HHt, n*n*sizeof(double)));
286 	CUDA_ERROR_CHECK(cudaMemset(d_HHt, 0, n*n*sizeof(double)));
287 	
288 	//cblas_dgemm(CblasColMajor,CblasNoTrans,CblasTrans,n,n,k,1,H,n,H,n,0,HHt,n);
289 	double alpha = 1;
290 	double beta = 0;
291 	status = cublasDgemm(handle, CUBLAS_OP_N, CUBLAS_OP_T, n, n, k, &alpha, d_H, n, d_H, n, &beta, d_HHt, n);CUBLAS_ERROR_CHECK(status);
292 	
293 	double* gradW;
294 	gradW = (double *)malloc(m*n*sizeof(double));
295 	memset(gradW, 0, m*n*sizeof(double));
296 
297 	double * d_gradW = 0;
298 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_gradW, m*n*sizeof(double)));
299 	CUDA_ERROR_CHECK(cudaMemset(d_gradW, 0, m*n*sizeof(double)));
300 	status = cublasDgemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, m, n, n, &alpha, d_W, m, d_HHt, n, &beta, d_gradW, m);CUBLAS_ERROR_CHECK(status);
301 	
302 	double *d_VHt = 0;
303 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_VHt, m*n*sizeof(double)));
304 	CUDA_ERROR_CHECK(cudaMemset(d_VHt, 0, m*n*sizeof(double)));
305 	status = cublasDgemm(handle, CUBLAS_OP_N, CUBLAS_OP_T, m, n, k, &alpha, d_V, m, d_H, n, &beta, d_VHt, m);CUBLAS_ERROR_CHECK(status);
306 	alpha = -1;
307 	status = cublasDaxpy(handle, m*n, &alpha, d_VHt, 1, d_gradW, 1);CUBLAS_ERROR_CHECK(status);
308 	
309 	double *d_WtW = 0;
310 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_WtW, n*n*sizeof(double)));
311 	CUDA_ERROR_CHECK(cudaMemset(d_WtW, 0, n*n*sizeof(double)));
312 	alpha = 1;
313 	status = cublasDgemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, n, n, m, &alpha, d_W, m, d_W, m, &beta, d_WtW, n);CUBLAS_ERROR_CHECK(status);
314 
315 	double* gradH;
316 	gradH = (double *)malloc(k*n*sizeof(double));
317 	memset(gradH, 0, k*n*sizeof(double));
318 
319 	double *d_gradH = 0;
320 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_gradH, n*k*sizeof(double)));
321 	CUDA_ERROR_CHECK(cudaMemset(d_gradH, 0, n*k*sizeof(double)));
322 	status = cublasDgemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, n, k, n, &alpha, d_WtW, n, d_H, n, &beta, d_gradH, n);CUBLAS_ERROR_CHECK(status);
323 	
324 	double *d_WtV = 0;
325 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_WtV, n*k*sizeof(double)));
326 	CUDA_ERROR_CHECK(cudaMemset(d_WtV, 0, n*k*sizeof(double)));
327 	status = cublasDgemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, n, k, m, &alpha, d_W, m, d_V, m, &beta, d_WtV, n);CUBLAS_ERROR_CHECK(status);
328 	alpha = -1;
329 	status = cublasDaxpy(handle, n*k, &alpha, d_WtV, 1, d_gradH, 1);CUBLAS_ERROR_CHECK(status);
330 	
331 	double initgrad = 0, tempgad = 0;
332 	status = cublasDdot (handle, m*n, d_gradW, 1, d_gradW, 1, &initgrad);CUBLAS_ERROR_CHECK(status);
333 	status = cublasDdot (handle, m*n, d_gradH, 1, d_gradH, 1, &tempgad);CUBLAS_ERROR_CHECK(status);
334 	initgrad = sqrt(initgrad + tempgad);
335 
336 	printf("Init gradient norm %f\n", initgrad); 
337 	double tolW = initgrad*max(0.001,tol); 
338 	double tolH = tolW;
339 	
340 	double *tmpvec = 0;
341 	tmpvec = (double *)malloc(m*n*sizeof(double));
342 	memset(tmpvec, 0, m*n*sizeof(double));
343 	
344 	double *tmpvec2 = 0;
345 	tmpvec2 = (double *)malloc(n*k*sizeof(double));
346 	memset(tmpvec2, 0, n*k*sizeof(double));
347 
348 	double *d_tmpvec = 0;
349 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_tmpvec, m*n*sizeof(double)));
350 	CUDA_ERROR_CHECK(cudaMemset(d_tmpvec, 0, m*n*sizeof(double)));
351 	
352 	double *d_tmpvec2 = 0;
353 	CUDA_ERROR_CHECK(cudaMalloc((void **)&d_tmpvec2, k*n*sizeof(double)));
354 	CUDA_ERROR_CHECK(cudaMemset(d_tmpvec2, 0, n*k*sizeof(double)));
355 	
356 	double *Vt = 0;		
357 	Vt = (double *)malloc(m*k*sizeof(double));
358 	memset(Vt, 0, m*k*sizeof(double));
359 	transpose(V, Vt, k, m);
360 
361 	double *Ht = 0;		
362 	Ht = (double *)malloc(n*k*sizeof(double));
363 	memset(Ht, 0, n*k*sizeof(double));
364 	
365 	double *Wt = 0;		
366 	Wt = (double *)malloc(m*n*sizeof(double));
367 	memset(Wt, 0, m*n*sizeof(double));
368 	
369 
370 	CUDA_ERROR_CHECK(cudaMemcpy(gradW, d_gradW, m*n*sizeof(double), cudaMemcpyDeviceToHost));
371 	CUDA_ERROR_CHECK(cudaMemcpy(gradH, d_gradH, k*n*sizeof(double), cudaMemcpyDeviceToHost));
372 
373 	double projnorm = 0;
374 	double projnorm2 = 0;
375 	int iter = 0;	
376 	for (iter = 1; iter <= maxiter; iter++){
377 		//projnorm = norm([gradW(gradW<0 | W>0); gradH(gradH<0 | H>0)]);
378 		
379 		int ii = 0;
380 		for (int i = 0; i < m*n; i++){
381 			if (gradW[i] < 0 || W[i] > 0 )
382 				tmpvec[ii++] = gradW[i];
383 		}
384 		CUDA_ERROR_CHECK(cudaMemcpy(d_tmpvec, tmpvec, ii*sizeof(double), cudaMemcpyHostToDevice));
385 		status = cublasDdot(handle, ii, d_tmpvec, 1, d_tmpvec, 1, &projnorm);CUBLAS_ERROR_CHECK(status);
386 
387 		ii = 0;
388 		for (int i = 0; i < n*k; i++){
389 			if (gradH[i] < 0 || H[i] > 0 )
390 				tmpvec2[ii++] = gradH[i];
391 		}
392 		CUDA_ERROR_CHECK(cudaMemcpy(d_tmpvec2, tmpvec2, ii*sizeof(double), cudaMemcpyHostToDevice));
393 		status = cublasDdot (handle, ii, d_tmpvec2, 1, d_tmpvec2, 1, &projnorm2);CUBLAS_ERROR_CHECK(status);
394 		projnorm = sqrt(projnorm + projnorm2);
395 		
396 		if ( projnorm < tol*initgrad || time(NULL) - initt > timelimit)
397 			break;
398 		
399 		transpose(H, Ht, n, k);	
400 		transpose(W, Wt, m, n);
401 
402 		//[W,gradW,iterW] = nlssubprob(V',H',W',tolW,1000);
403 		int iterW = 0;
404 		nlssubprob(Vt, Ht, Wt, k, n, m, tolW, 1000, W, gradW, &iterW);
405 	
406 		transpose(W, W, m, n);
407 		transpose(gradW, gradW, m, n);
408 
409 		if (iterW == 1)
410 			tolW = 0.1 * tolW;
411 		
412 		int iterH = 0;
413 		nlssubprob(V, W, H, m, n, k, tolH, 1000, H, gradH, &iterH);
414 		if (iterH == 1)
415 			tolH = 0.1 * tolH;
416 		
417 		if ( iter%10 == 0)
418 			printf(".");
419 	}
420 	CUDA_ERROR_CHECK(cudaMemcpy(d_W, W, m*n*sizeof(double), cudaMemcpyHostToDevice));
421 	CUDA_ERROR_CHECK(cudaMemcpy(d_H, H, k*n*sizeof(double), cudaMemcpyHostToDevice));
422 	printf("\nIter = %d Final proj-grad norm %f\n", iter, projnorm);
423
